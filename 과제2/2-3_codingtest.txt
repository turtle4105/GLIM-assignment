#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    
    int minX = dots[0][0];
    int maxX = dots[0][0];
    int minY = dots[0][1];
    int maxY = dots[0][1];

    int n = dots.size();
    for (int i = 1; i < n; i++) {
        int x = dots[i][0];
        int y = dots[i][1];

        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
    }

    int width  = maxX - minX;
    int height = maxY - minY;
    
    return width * height; 
}
