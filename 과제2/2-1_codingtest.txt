#include <string>
#include <vector>

using namespace std;

const int LAMB_PRICE = 12000;
const int DRINK_PRICE = 2000;

int solution(int n, int k) {
    int service = n / 10;
    int payDrink = 0;
    
    if (k > service) payDrink = k - service;
    
    return LAMB_PRICE * n + (DRINK_PRICE * payDrink);
}
